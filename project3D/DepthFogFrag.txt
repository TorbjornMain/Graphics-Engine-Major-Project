#version 410

in vec2 uvs;

uniform sampler2D screen;
uniform sampler2D depthBuffer;

out vec4 fragColor;

void main()
{
	float scsp = clamp(pow(texture(depthBuffer, uvs).x, 7), 0, 1);

	fragColor = vec4(scsp * 0.05, scsp * 0.75, scsp, 1);
	fragColor += (1 - scsp) * texture(screen, uvs); // + vec4(0.1f, 0, 0, 0);
	fragColor = texture(screen, uvs);
}
