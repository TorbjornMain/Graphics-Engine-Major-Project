#version 410
layout(location=0) in vec3 position;
layout(location=1) in vec3 normal;
layout(location=2) in vec2 uvs; 

uniform mat4 PVW;

uniform mat4 M;

uniform float time;

out vec4 worldNorm;
out vec4 worldPos;
out vec2 uv;

void main() {  
	vec3 P = position;
	uv = uvs;
	worldNorm = (M * vec4(normal,0));
	worldPos = (M*vec4(P,1));
	gl_Position = PVW * vec4(P, 1);
}
